let peer; let myStream; const RECEIVER_ID = "dan1982"; // The PC's fixed ID async function startApp(role) { document.getElementById('menu').style.display = 'none'; document.getElementById('app').style.display = 'block'; const peerOptions = { debug: 2, config: {'iceServers': [ { 'urls': 'stun:stun.l.google.com:19302' }, { 'urls': 'stun:stun1.l.google.com:19302' } ]} }; // PC uses the fixed ID, Phone gets a random ID peer = (role === 'receiver') ? new Peer(RECEIVER_ID, peerOptions) : new Peer(peerOptions); peer.on('open', id => { const status = document.getElementById('status'); console.log("My peer ID is: " + id); status.innerText = (role === 'receiver') ? "PC READY: Awaiting Phone..." : "PHONE READY"; status.style.color = "#00ff00"; if(role === 'sender') document.getElementById('controls').style.display = 'block'; }); // PHONE SIDE: Get Camera if (role === 'sender') { try { myStream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" }, audio: true }); document.getElementById('v').srcObject = myStream; } catch (e) { alert("Camera access denied! Check HTTPS and Permissions."); } } // PC SIDE: Listen for the call peer.on('call', call => { console.log("Receiving call..."); call.answer(); // PC answers with nothing (or its own stream) call.on('stream', remoteStream => { const video = document.getElementById('v'); video.srcObject = remoteStream; // Browsers often block auto-playing audio. We play it manually: video.play(); document.getElementById('status').innerText = "LIVE: CONNECTED"; }); }); peer.on('error', err => { console.error("PeerJS Error:", err.type); document.getElementById('status').innerText = "Error: " + err.type; document.getElementById('status').style.borderColor = "red"; }); } function connect() { if (!myStream) { alert("No camera stream found!"); return; } document.getElementById('status').innerText = "Calling PC..."; const call = peer.call(RECEIVER_ID, myStream); call.on('error', err => { alert("Link failed: " + err); }); }
